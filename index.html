// Example: Fetching AI Response
const prompt = "Write a hello world function in C++";
const endpoint = `https://r-gengpt-api.vercel.app/api?prompt=${encodeURIComponent(prompt)}`;

async function getResponse() {
  try {
    const response = await fetch(endpoint);
    const data = await response.text();
    console.log("R-GenGPT:", data);
  } catch (err) {
    console.error(err);
  }
}

getResponse();
# Example: Using Requests Library
import requests
import urllib.parse

prompt = "Explain Quantum Computing"
encoded_prompt = urllib.parse.quote(prompt)
url = f"https://r-gengpt-api.vercel.app/api?prompt={encoded_prompt}"

try:
    response = requests.get(url)
    print("R-GenGPT Response:")
    print(response.text)
except Exception as e:
    print(e)
